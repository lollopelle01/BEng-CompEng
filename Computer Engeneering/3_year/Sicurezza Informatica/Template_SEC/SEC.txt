• Protezione comunicazione
	openssl
		openssl enc -aes-256-cbc -md sha512 -pbkdf2 -iter 100000 -salt -in file_testo.txt -out cifrato.bin —> cifrare file
		openssl enc -aes-256-cbc -md sha512 -pbkdf2 -iter 100000 -salt -d in file_testo.txt -out cifrato.bin —> decifrare file
		openssl genrsa -out chiave.pem 2048 —> generazione chiave privata
		openssl rsa -in chiave.pem -text -noout —> info chiave privata
		openssl rsa -in chiave.pem -aes-256-cbc -out enc_chiave.pem —> crittare un file (?)
		openssl rsa -in chiave.pem -pubout -out pub_chiave.pem —> estrazione chiave pubblica
		openssl rsautl -encrypt -in testo.txt -inkey chiave.pem -out cifrato_rsa.bin —> crittare file (?)
		openssl rsautl -decrypt -in cifrato_rsa.bin -inkey chiave.pem -out nuova.txt —> decifrare file (?)
