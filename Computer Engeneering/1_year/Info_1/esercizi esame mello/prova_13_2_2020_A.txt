#include <stdio.h>
#include "list.h"

list testLists(list p, list q, int n){
	if(n==0)
		return emptylist();
	if(n>0){
		if(head(p)>=head(q))
			return cons(head(p), testLists(tail(p), tail(q), n);
		if(head(p)<=head(q))
			return cons(head(q), testLists(tail(p), tail(q), n);
	}
	if(n<0){
		if(head(p)<=head(q))
			return cons(head(p), testLists(tail(p), tail(q), n);
		if(head(p)>=head(q))
			return cons(head(q), testLists(tail(p), tail(q), n);
	}
}

int main(){
	list p, q, r1, r2, r3;
	q=cons(-1, cons(0, cons(5, emptylist())));
	p=cons(-1, cons(4, cons(3, emptylist())));
	r1=testLists(p, q, 3);
	r2=testLists(p, q, -2);
	r3=testLists(p, q, 0);
}
