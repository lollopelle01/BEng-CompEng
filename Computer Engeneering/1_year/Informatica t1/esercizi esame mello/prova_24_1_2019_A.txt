#include <stdio.h>
#include "list.h"

list elemSum(list l, int elem){
	if(empty(l))
		return emptylist();
	else
		return cons(head(l)+elem, elemSum(tail(l), elem));
}

int lastList(list l){
	int K;
	while(!empty(l)){
		K=head(l);
		l=tail(l);
	}
	return K
}

int main(){
	int E;
	list l1, l2;
	l1=cons(2, cons(0, cons(5, cons(-1, cons(3, emptylist())))));
	E=lastList(l1);
	l2=elemSum(l1, E);
	
	while(!empty(l2)){
		printf("%d\n", head(l2));
		l2=tail(l2);
	}
	
	return 0;
}
