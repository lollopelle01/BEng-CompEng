#include <stdio.h>
#include "list.h"

list newList(list p, int n){
	if(n<=0)
		return p;
	else{
		if(n==1)
			return tail(p);
		else
			return (head(p), newList(tail(p), n-1);
	}

}

int main(){
	list l1, p=cons(7, cons(3, cons(5, emptylist())));
	int n;
	scanf("%d", &n);
	l1=newList(p,n);
	
	while(!empty(l1)){
		printf("%d\n", head(l1));
		l1=tail(l1);
	}
	return 0;
}
